\begin{frame}{Teste x Debug}
	\begin{itemize}
		\item Testar e Debugar é a mesma coisa?
		\item a Resposta é...
		\item Não.
		\item Teste objetiva revelar a presença de defeitos.
		\item Debug é o processo de encontrar o corrigir os defeitos.
	\end{itemize}
\end{frame}

%-------------------------------------------------------------------------Slide

\begin{frame}{Teste x Homologação}
	\begin{itemize}
		\item Testar e homologar é a mesma coisa?
		\item a Resposta é...
		\item Não.
		\item os objetivos são diferentes.
		\item Teste objetiva revelar a presença de defeitos.
		\item Homologar objetiva aferir que o funcionamento esteja adequado às necessidades explícitadas.
	\end{itemize}
\end{frame}

\begin{frame}{Tipos de Teste}
  \begin{itemize}
   \item Diferentes tipos de testes podem ser utilizados para verificar se um programa se comporta como o especificado.
   \item Basicamente, os testes podem ser classificados em:
   \begin{itemize}
    \item Teste Funcional (\textbf{teste caixa-preta});
    \item Teste Estrutural (\textbf{teste caixa-branca}) e 
    \item Teste Baseado em Defeitos (\textbf{Testes de Mutantes}).
   \end{itemize}
  \end{itemize}
\end{frame}

%-------------------------------------------------------------------------Slide

\begin{frame}{Tipos de Teste}
  \begin{itemize}
   \item A t\'ecnica de teste \'e definida pelo tipo de informa\c{c}\~ao utilizada para realizar o teste.
  \end{itemize}
\end{frame}

%-------------------------------------------------------------------------Slide

\begin{frame}{Técnica de Teste Funcional}
  \begin{itemize}
    \item \textbf{T\'ecnica caixa-preta} - os testes s\~ao baseados exclusivamente na especifica\c{c}\~ao de requisitos do programa. Nenhum conhecimento de como o programa est\'a implementado \'e requerido.
  \end{itemize}
  \begin{figure}
   \centering
   \includegraphics[width=5cm]{testeCaixaPreta}
   \caption[numbered]\insertcaptionnumber{ - Técnica de Teste Funcional. \cite{neto2009}}
  \end{figure}
\end{frame}

%-------------------------------------------------------------------------Slide

\begin{frame}{Técnica de Teste Estrutural}
  \begin{itemize}
    \item \textbf{T\'ecnica caixa-branca} - os testes s\~ao baseados na estrutura interna do programa, ou seja, na implementa\c{c}\~ao do mesmo.
  \end{itemize}
  \begin{figure}
   \centering

\begin{frame}{Níveis de Cobertura}
	\begin{itemize}
		\item Diferentes níveis de cobertura podem ser definidos em função dos elementos do GFC.
		\item Cobertura: porcentagem dos requisitos que foram testados versus o total de requisitos gerados.
		\item Oito diferentes n??veis de cobertura são definidos por \cite{Copeland2003}.	
		\item Quanto maior o nível, maior o rigor do critério de teste, ou seja, mais caso de teste ele exige para ser satisfeito.
\begin{itemize}
	\item Nível 0 $\longleftarrow$ Nível 1 $\longleftarrow$ Nível 2 $\longleftarrow$ Nível 3 $\longleftarrow$ Nível 4 $\longleftarrow$ Nível 5 $\longleftarrow$ Nível 6 $\longleftarrow$ Nível 7

\begin{frame}{Níveis de Cobertura}
	\begin{itemize}
		\item Nível 0: qualquer valor de cobertura inferior a 100\% da cobertura de todos os comandos.
		\item Nível 1: 100\% de cobertura de comandos. (Todos-Nós)
		\item Nível 2: 100\% de cobertura de decisões. (Todos-Arcos)
		\item Nível 3: 100\% de cobertura de condições.
		\item Nível 4: 100\% de cobertura decisões/condições. 
		\item Nível 5: 100\% de cobertura de condições múltiplas.
		\item Nível 6: cobertura de loop.
		\item Nível 7: 100\% de cobertura de caminhos. (Todos-Caminhos)							
	\end{itemize}
\end{frame}
