  \item Indenpendentemente do tipo do atributo:

    \begin{itemize}

\begin{frame}{Particionamento por Equival\^encia - Diretrizes}
 \begin{itemize}
  \item Atributos do tipo String:
  \begin{itemize}
  \item Acentuação gr\'afica / Caracteres especiais (sim ou não)
  \begin{itemize}
   \item Definir uma classe para o ``sim'';
   \item Definir uma classe para o ``n\~ao''
  \end{itemize}
  \item Tamanho mínimo / Tamanho máximo
  \begin{itemize}
   \item Definir uma classe inv\'alida para valor inferior ao m\'inimo;
   \item Definir uma classe v\'alida para valor dentro do intervalo entre o m\'inimo e o m\'aximo.
   \item Definir uma classe inv\'alida para valor superior do m\'aximo.
  \end{itemize}
  \end{itemize}
 \end{itemize}
\end{frame}

  \item Atributos de tipos num\'ericos
  \begin{itemize}
   \item Valores Inteiros:
    \begin{itemize}
     \item Definir uma classe inv\'alida para valores inferiores ao limite inferior;
     \item Definir uma classe v\'alida para valores dentro do intervalo entre os limites;
     \item Definir uma classe inv\'alida para valores acima do limite superior. 
    \end{itemize}
   \item Valores de ponto-flutuante:
   \begin{itemize}
    \item Definir tr\^es classes de equival\^encia para a quantidade de casas depois da v\'irgula.
   \end{itemize}
  \end{itemize}
 \end{itemize}
\end{frame}

\begin{frame}{Particionamento por Equival\^encia - Diretrizes}
 \begin{itemize}
  \item Atributos do tipo l\'ogico
  \begin{itemize}
   \item Definir duas classes de equival\^encia.
  \end{itemize}
  \item Atributos do tipo Data:
   \begin{itemize}
    \item Definir classes de equival\^encia para:
    \begin{itemize}
     \item Data V\'alida e inv\'alida;
     \item Formato
     \item Data inferior e data superior.
    \end{itemize}
   \end{itemize}
  \end{itemize}
\end{frame}
